var vMinXP=100;
var vMaxAtt=120;

// Items

var vMPPotionName="mpot0";

// Support Info
var vSupportName="RoseMoth";
var vSupportDistance;

// Skill Variables
vTaunt="taunt";

// Load External Code Block

// Start Character Loop
setInterval(function(){
	
	//set potion values for future function use
	var vHPPotionValue=fFindPotionValue(vHPPotionName);
	var vMPPotionValue=fFindPotionValue(vMPPotionName);
	
	// Functions
	
	// Support Info
	vSupportDistance=fSupportDistance(vSupportName);	
	
	
	// Potions / Regen
	fUseHPPotion(vHPPotionValue,vSupportDistance);
	fUseMPPotion(vMPPotionValue);
	fUseHPRegen(vHPPotionValue,vSupportDistance);
	fUseMPRegen(vMPPotionValue);
	
	// Attack
	fNextTarget();
	fGoToTarget();
	fSkillOnAllyTargeted(vSupportName,vCharge);
	fSkillOnAllyTargeted(vSupportName,vTaunt);
	fAttackLoop(vAttackMode,vMinXP,vMaxAtt);

	
	



},1000/4); // Loops every 1/4 seconds.
